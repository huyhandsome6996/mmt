# Tính Đóng Gói: Giấu dữ liệu (Private) 
# và dùng Getter/Setter để kiểm soát.

class TaiKhoan:
    def __init__(self, chu_tk, so_du):
        self.chu_tk = chu_tk
        # Dùng 2 dấu gạch dưới (__) để làm biến Private
        self.__so_du = so_du  

    # GETTER: Lấy dữ liệu ra an toàn
    @property
    def so_du(self):
        return self.__so_du

    # SETTER: Cập nhật dữ liệu có kiểm soát
    @so_du.setter
    def so_du(self, so_tien):
        if so_tien >= 0:
            self.__so_du = so_tien
        else:
            print("Lỗi: Tiền không hợp lệ!")

# --- Chạy thử ---
tk = TaiKhoan("An", 500)

# Cố tình truy cập biến private -> LỖI AttributeError
# print(tk.__so_du) 

# Truy cập qua Getter/Setter -> THÀNH CÔNG
tk.so_du = 1000 # Gọi setter
print(tk.so_du) # Gọi getter

KếThừa.py 
# Tính Kế Thừa: Lớp con xài lại đồ của lớp cha.

# LỚP CHA
class NhanVien:
    def __init__(self, ten, luong_cb):
        self.ten = ten
        self.luong_cb = luong_cb
        
    def lam_viec(self):
        print(f"{self.ten} đang làm việc.")

# LỚP CON (Bỏ tên lớp cha vào ngoặc tròn)
class QuanLy(NhanVien):
    def __init__(self, ten, luong_cb, thuong):
        # Dùng super() để ném Tên và Lương lên cho cha xử lý
        super().__init__(ten, luong_cb)
        
        # Bổ tương đặc điểm riêng của lớp con
        self.thuong = thuong
        
    # Thêm hành vi riêng của Quản Lý
    def hop_nhan_vien(self):
        print(f"Sếp {self.ten} đang họp!")

# --- Chạy thử ---
sep = QuanLy("Châu", 10000, 5000)

# Xài hàm của lớp cha
sep.lam_viec() 

# Xài hàm của chính nó
sep.hop_nhan_vien()

using System;

// Tính Kế Thừa: Lớp con xài lại đồ của lớp cha.

// LỚP CHA
public class NhanVien
{
    public string Ten { get; set; }
    public double LuongCB { get; set; }

    public NhanVien(string ten, double luongCb) {
        Ten = ten;
        LuongCB = luongCb;
    }

    public void LamViec() {
        Console.WriteLine($"{Ten} đang làm việc.");
    }
}

// LỚP CON (Dùng dấu 2 chấm ':')
public class QuanLy : NhanVien
{
    public double Thuong { get; set; }

    // Dùng ': base()' để ném dữ liệu lên cho hàm tạo của cha
    public QuanLy(string ten, double luong, double thuong) 
        : base(ten, luong)
    {
        Thuong = thuong;
    }

    // Thêm hành vi riêng
    public void HopNhanVien() {
        Console.WriteLine($"Sếp {Ten} đang họp!");
    }
}

// --- Chạy thử ---
class Program {
    static void Main() {
        QuanLy sep = new QuanLy("Châu", 10000, 5000);
        
        sep.LamViec();     // Của lớp cha
        sep.HopNhanVien(); // Của riêng nó
    }
}

# Tính Đa Hình (Polymorphism):
# Cùng 1 tên hàm, nhưng mỗi đối tượng làm 1 kiểu khác nhau.

class NhanVien:
    def __init__(self, ten, luong_cb):
        self.ten = ten
        self.luong_cb = luong_cb

    # Hàm chung
    def tinh_luong(self):
        return self.luong_cb

class QuanLy(NhanVien):
    def __init__(self, ten, luong_cb, thuong):
        super().__init__(ten, luong_cb)
        self.thuong = thuong

    # GHI ĐÈ (Override) hàm của cha để tính theo kiểu khác
    def tinh_luong(self):
        return self.luong_cb + self.thuong

# --- Chạy thử ---
nv1 = NhanVien("An", 5000)
nv2 = QuanLy("Châu", 5000, 2000)

danh_sach = [nv1, nv2]

# Duyệt mảng. Cùng gọi hàm tinh_luong()
# nhưng máy tự biết ai tính kiểu thường, ai tính kiểu sếp.
for nv in danh_sach:
    print(f"{nv.ten}: {nv.tinh_luong()} VNĐ")
DaHinh.cs 
using System;
using System.Collections.Generic;

// Tính Đa Hình: C# khắt khe hơn.
// Phải xin phép (virtual) và xác nhận (override).

public class NhanVien
{
    public string Ten { get; set; }
    public double LuongCB { get; set; }

    public NhanVien(string ten, double luongCb) {
        Ten = ten; LuongCB = luongCb;
    }

    // Từ khóa VIRTUAL: Cho phép lớp con được quyền ghi đè
    public virtual double TinhLuong() {
        return LuongCB;
    }
}

public class QuanLy : NhanVien
{
    public double Thuong { get; set; }

    public QuanLy(string ten, double luong, double thuong) 
        : base(ten, luong) { Thuong = thuong; }

    // Từ khóa OVERRIDE: Chấp nhận ghi đè hàm của cha
    public override double TinhLuong() {
        return LuongCB + Thuong;
    }
}

class Program {
    static void Main() {
        // Gom chung vào 1 List kiểu NhanVien
        List ds = new List {
            new NhanVien("An", 5000),
            new QuanLy("Châu", 5000, 2000)
        };

        foreach (var nv in ds) {
            Console.WriteLine($"{nv.Ten}: {nv.TinhLuong()} VNĐ");
        }
    }
}
TrừuTượng.py 
# Tính Trừu Tượng: Tạo ra "Bản Hợp Đồng".
# Bắt buộc lớp con phải làm theo, nếu không sẽ báo lỗi.

from abc import ABC, abstractmethod

# LỚP TRỪU TƯỢNG (Kế thừa từ ABC)
# Lớp này không thể khởi tạo đối tượng trực tiếp.
class HinhHoc(ABC):
    def __init__(self, ten):
        self.ten = ten

    # HÀM TRỪU TƯỢNG
    # Lớp cha chỉ nêu Tên Hàm, không làm gì cả (pass)
    # ÉP BẮT lớp con phải tự điền ruột cho hàm này!
    @abstractmethod
    def tinh_dien_tich(self):
        pass 

# LỚP CON
class HinhVuong(HinhHoc):
    def __init__(self, canh):
        super().__init__("Hình Vuông")
        self.canh = canh

    # Nếu HinhVuong quên viết hàm tinh_dien_tich() này, 
    # Python sẽ CHỬI LỖI ngay khi bấm Run!
    def tinh_dien_tich(self):
        return self.canh * self.canh

# --- Chạy thử ---
# loi = HinhHoc("Lỗi") # Lỗi! Không thể tạo ĐT từ Lớp Trừu tượng

h1 = HinhVuong(4)
print(h1.tinh_dien_tich())
TruuTuong.cs 
using System;

// Tính Trừu Tượng: Tạo ra "Bản Hợp Đồng".

// Bắt buộc dùng từ khóa 'abstract' ở tên Class
public abstract class HinhHoc
{
    public string Ten { get; set; }
    public HinhHoc(string ten) { Ten = ten; }

    // HÀM TRỪU TƯỢNG (Dùng từ khóa abstract)
    // Chấm phẩy luôn, KHÔNG CÓ NGOẶC NHỌN { } !
    public abstract double TinhDienTich();
}

// LỚP CON
public class HinhVuong : HinhHoc
{
    public double Canh { get; set; }

    public HinhVuong(double canh) : base("Hình Vuông") {
        Canh = canh;
    }

    // C# bắt buộc Lớp Con phải dùng chữ 'override' 
    // để xác nhận đang trả nợ bản hợp đồng cho Lớp Cha.
    public override double TinhDienTich()
    {
        return Canh * Canh;
    }
}

// --- Chạy thử ---
class Program {
    static void Main() {
        // HinhHoc loi = new HinhHoc("Lỗi"); // Báo lỗi đỏ ngay!

        HinhVuong h1 = new HinhVuong(4);
        Console.WriteLine(h1.TinhDienTich());
    }
}

# TỪ KHÓA 'self' TRONG PYTHON
# self đại diện cho chính đối tượng đang gọi hàm.

class Computer:
    # 1. BẮT BUỘC có 'self' làm tham số đầu tiên 
    # ở MỌI phương thức (kể cả hàm tạo).
    def __init__(self, gia_ban):
        
        # 2. BẮT BUỘC dùng 'self.' để tạo và truy cập 
        # thuộc tính của class. 
        # Nếu chỉ ghi: gia_ban = gia_ban (Sẽ bị tính là biến cục bộ)
        self.gia_ban = gia_ban

    # Tham số self tự động được truyền ngầm khi ta gọi đối tượng
    def hien_thi(self):
        # 3. Muốn xài đồ của class, phải gọi qua self
        print(f"Giá máy tính: {self.gia_ban}")

# --- Chạy thử ---
may1 = Computer(1500)

# Thực chất, Python ngầm hiểu là: Computer.hien_thi(may1)
# Đối tượng 'may1' được ném thẳng vào chữ 'self'
may1.hien_thi()

using System;

// TỪ KHÓA 'this' TRONG C#
// Tương đương self, nhưng C# đủ thông minh để tự ngầm hiểu.

public class Computer
{
    private double giaBan; // Khai báo thuộc tính ở ngoài hàm

    // 1. KHÔNG BAO GIỜ viết 'this' vào tham số của hàm
    public Computer(double giaBan) 
    {
        // 2. CHỈ BẮT BUỘC dùng 'this.' khi tên tham số bị TRÙNG 
        // với tên thuộc tính của class (như chữ giaBan ở trên).
        // this.giaBan (Tài sản của class) = giaBan (Biến truyền vào)
        this.giaBan = giaBan;
    }

    // Không cần ghi this vào tham số
    public void HienThi() 
    {
        // 3. KHÔNG BẮT BUỘC dùng 'this.' ở đây
        // Do không có biến nào trùng tên, C# tự biết giaBan 
        // là tài sản của class.
        Console.WriteLine($"Giá máy tính: {giaBan}");
        
        // (Viết this.giaBan ở đây cũng không sai, nhưng thường lược bỏ)
    }
}

class Program {
    static void Main() {
        Computer may1 = new Computer(1500);
        may1.HienThi();
    }
}

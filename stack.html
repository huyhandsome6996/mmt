def decimal_to_binary(dec_num):
s = Stack()
if dec_num == 0:
return "0"
while dec_num > 0:
remainder = dec_num % 2
s.push(remainder)
dec_num = dec_num // 2
bin_string = ""
while not s.is_empty():
bin_string += str(s.pop())
return bin_string
